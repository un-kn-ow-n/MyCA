 40 /**
 41  * kjur's ASN.1 class for RFC 5126 CAdES long term signature
 42  * <p>
 43  * This name space provides 
 44  * <a href="https://tools.ietf.org/html/rfc5126">RFC 5126
 45  * CAdES(CMS Advanced Electronic Signature)</a> generator.
 46  *
 47  * <h4>SUPPORTED FORMATS</h4>
 48  * Following CAdES formats is supported by this library.
 49  * <ul>
 50  * <li>CAdES-BES - CAdES Basic Electronic Signature</li>
 51  * <li>CAdES-EPES - CAdES Explicit Policy-based Electronic Signature</li>
 52  * <li>CAdES-T - Electronic Signature with Time</li>
 53  * </ul>
 54  * </p>
 55  *
 56  * <h4>PROVIDED ATTRIBUTE CLASSES</h4>
 57  * <ul>
 58  * <li>{@link KJUR.asn1.cades.SignaturePolicyIdentifier} - for CAdES-EPES
 59  *   <ul>
 60  *   <li>{@link KJUR.asn1.cades.SignaturePolicyId}</li>
 61  *   </ul>
 62  * </li>
 63  * <li>{@link KJUR.asn1.cades.SignatureTimeStamp} - for CAdES-T</li>
 64  * <li>{@link KJUR.asn1.cades.CompleteCertificateRefs} - for CAdES-C(for future use)
 65  *   <ul>
 66  *   <li>{@link KJUR.asn1.cades.OtherCertID}</li>
 67  *   <li>{@link KJUR.asn1.cades.OtherHash}</li>
 68  *   <li>{@link KJUR.asn1.cades.OtherHashAlgAndValue}</li>
 69  *   <li>{@link KJUR.asn1.cades.OtherHashValue}</li>
 70  *   </ul>
 71  * </li>
 72  * </ul>
 73  * NOTE: Currntly CAdES-C is not supported since parser can't
 74  * handle unsigned attribute.
 75  * 
 76  * <h4>OTHER CLASSES</h4>
 77  * <ul>
 78  * <li>{@link KJUR.asn1.cades.CAdESUtil} - utilities for CAdES</li>
 79  * </ul>
 80  *
 81  * <h4>GENERATE CAdES-BES</h4>
 82  * To generate CAdES-BES, {@link KJUR.asn.cades} namespace 
 83  * classes are not required and already {@link KJUR.asn.cms} namespace 
 84  * provides attributes for CAdES-BES.
 85  * Create {@link KJUR.asn1.cms.SignedData} with following
 86  * mandatory attribute in CAdES-BES:
 87  * <ul>
 88  * <li>{@link KJUR.asn1.cms.ContentType}</li>
 89  * <li>{@link KJUR.asn1.cms.MessageDigest}</li>
 90  * <li>{@link KJUR.asn1.cms.SigningCertificate} or </li>
 91  * <li>{@link KJUR.asn1.cms.SigningCertificateV2}</li>
 92  * </ul>
 93  * CMSUtil.newSignedData method is very useful to generate CAdES-BES.
 94  * <pre>
 95  * sd = KJUR.asn1.cms.CMSUtil.newSignedData({
 96  *   content: {str: "aaa"},
 97  *   certs: [certPEM],
 98  *   signerInfos: [{
 99  *     hashAlg: 'sha256',
100  *     sAttr: {SigningCertificateV2: {array: [certPEM]}},
101  *     signerCert: certPEM,
102  *     sigAlg: 'SHA256withRSA',
103  *     signerPrvKey: pkcs8PrvKeyPEM
104  *   }]
105  * });
106  * signedDataHex = sd.getContentInfoEncodedHex();
107  * </pre>
108  * NOTE: ContentType and MessageDigest signed attributes
109  * are automatically added by default.
110  *
111  * <h4>GENERATE CAdES-BES with multiple signers</h4>
112  * If you need signature by multiple signers, you can 
113  * specify one or more items in 'signerInfos' property as below.
114  * <pre>
115  * sd = KJUR.asn1.cms.CMSUtil.newSignedData({
116  *   content: {str: "aaa"},
117  *   certs: [certPEM1, certPEM2],
118  *   signerInfos: [{
119  *     hashAlg: 'sha256',
120  *     sAttr: {SigningCertificateV2: {array: [certPEM1]}},
121  *     signerCert: certPEM1,
122  *     sigAlg: 'SHA256withRSA',
123  *     signerPrvKey: pkcs8PrvKeyPEM1
124  *   },{
125  *     hashAlg: 'sha1',
126  *     sAttr: {SigningCertificateV2: {array: [certPEM2]}},
127  *     signerCert: certPEM2,
128  *     sigAlg: 'SHA1withRSA',
129  *     signerPrvKey: pkcs8PrvKeyPEM2
130  *   }]
131  * });
132  * signedDataHex = sd.getContentInfoEncodedHex();
133  * </pre>
134  *
135  * <h4>GENERATE CAdES-EPES</h4>
136  * When you need a CAdES-EPES signature,
137  * you just need to add 'SignaturePolicyIdentifier'
138  * attribute as below.
139  * <pre>
140  * sd = KJUR.asn1.cms.CMSUtil.newSignedData({
141  *   content: {str: "aaa"},
142  *   certs: [certPEM],
143  *   signerInfos: [{
144  *     hashAlg: 'sha256',
145  *     sAttr: {
146  *       SigningCertificateV2: {array: [certPEM]},
147  *       SignaturePolicyIdentifier: {
148  *         oid: '1.2.3.4.5',
149  *         hash: {alg: 'sha1', hash: 'b1b2b3b4b...'}
150  *       },
151  *     },
152  *     signerCert: certPEM,
153  *     sigAlg: 'SHA256withRSA',
154  *     signerPrvKey: pkcs8PrvKeyPEM
155  *   }]
156  * });
157  * signedDataHex = sd.getContentInfoEncodedHex();
158  * </pre>
159  *
160  * <h4>GENERATE CAdES-T</h4>
161  * After a signed CAdES-BES or CAdES-EPES signature have been generated,
162  * you can generate CAdES-T by adding SigningTimeStamp unsigned attribute.
163  * <pre>
164  * beshex = "30..."; // hex of CAdES-BES or EPES data 
165  * info = KJUR.asn1.cades.CAdESUtil.parseSignedDataForAddingUnsigned(beshex);
166  * // You can refer a hexadecimal string of signature value 
167  * // in the first signerInfo in the CAdES-BES/EPES with a variable:
168  * // 'info.si[0].sigval'. You need to get RFC 3161 TimeStampToken
169  * // from a trusted time stamp authority. Otherwise you can also 
170  * // get it by 'KJUR.asn1.tsp' module. We suppose that we could 
171  * // get proper time stamp.
172  * tsthex0 = "30..."; // hex of TimeStampToken for signerInfo[0] sigval
173  * si0 = info.obj.signerInfoList[0];
174  * si0.addUnsigned(new KJUR.asn1.cades.SignatureTimeStamp({tst: tsthex0});
175  * esthex = info.obj.getContentInfoEncodedHex(); // CAdES-T
176  * </pre>
177  * </p>
178  *
179  * <h4>SAMPLE CODES</h4>
180  * <ul>
181  * <li><a href="../../tool_cades.html">demo program for CAdES-BES/EPES/T generation</a></li>
182  * <li><a href="../../test/qunit-do-asn1cades.html">Unit test code for KJUR.asn1.cades package</a></li>
183  * <li><a href="../../test/qunit-do-asn1tsp.html">Unit test code for KJUR.asn1.tsp package (See SimpleTSAAdaptor test)</a></li>
184  * <li><a href="../../test/qunit-do-asn1cms.html">Unit test code for KJUR.asn1.cms package (See newSignedData test)</a></li>
185  * </ul>
186  * 
187  * @name KJUR.asn1.cades
188  * @namespace
189  */
190 if (typeof KJUR.asn1.cades == "undefined" || !KJUR.asn1.cades) KJUR.asn1.cades = {};

192 /**
193  * class for RFC 5126 CAdES SignaturePolicyIdentifier attribute
194  * @name KJUR.asn1.cades.SignaturePolicyIdentifier
195  * @class class for RFC 5126 CAdES SignaturePolicyIdentifier attribute
196  * @param {Array} params associative array of parameters
197  * @extends KJUR.asn1.cms.Attribute
198  * @since jsrsasign 4.7.0 asn1cades 1.0.0
199  * @see KJUR.asn1.cms.AttributeList
200  * @see KJUR.asn1.cades.SignaturePolicyId
201  * @see KJUR.asn1.cades.OtherHashAlgAndValue
202  *
203  * @description
204  * This class provides ASN.1 encoder for
205  * <a href="https://tools.ietf.org/html/rfc5126#section-5.8.1">
206  * SignaturePolicyIdentifier defined in RFC 5126 CAdES section 5.8.1</a>.
207  * <pre>
208  * SignaturePolicyIdentifier ::= CHOICE {
209  *    signaturePolicyId       SignaturePolicyId,
210  *    signaturePolicyImplied  SignaturePolicyImplied } -- not used
211  *
212  * SignaturePolicyImplied ::= NULL
213  * SignaturePolicyId ::= SEQUENCE {
214  *    sigPolicyId           SigPolicyId,
215  *    sigPolicyHash         SigPolicyHash,
216  *    sigPolicyQualifiers   SEQUENCE SIZE (1..MAX) OF
217  *                             SigPolicyQualifierInfo OPTIONAL }
218  * SigPolicyId ::= OBJECT IDENTIFIER
219  * SigPolicyHash ::= OtherHashAlgAndValue
220  * </pre>
221  *
222  * @example
223  * new KJUR.asn1.cades.SignaturePolicyIdentifier({
224  *   attr: "signaturePolicyIdentifier",
225  *   oid: '1.2.3.4.5',
226  *   alg: 'sha1',
227  *   hash: 'a1a2a3a4...'
228  * })
229  */
230 KJUR.asn1.cades.SignaturePolicyIdentifier = function(params) {
231     var _KJUR = KJUR,
232 	_KJUR_asn1 = _KJUR.asn1,
233 	_KJUR_asn1_cades = _KJUR_asn1.cades,
234 	_SignaturePolicyId = _KJUR_asn1_cades.SignaturePolicyId;
235 	
236     _KJUR_asn1_cades.SignaturePolicyIdentifier.superclass.constructor.call(this);
237 
238     this.typeOid = "1.2.840.113549.1.9.16.2.15";
239 
240     this.params = null;
241 
242     this.getValueArray = function() {
243 	return [new _SignaturePolicyId(this.params)];
244     };
245 
246     this.setByParam = function(params) {
247 	this.params = params;
248     };
249 
250     if (params != undefined) this.setByParam(params);
251 };

257  * @name KJUR.asn1.cades.SignaturePolicyId

259  * @param {Array} params JSON object of parameters
260  * @extends KJUR.asn1.ASN1Object
261  * @since jsrsasign 10.0.0 asn1cades 2.0.0

328 /**
329  * class for OtherHashAlgAndValue ASN.1 object<br/>
330  * @name KJUR.asn1.cades.OtherHashAlgAndValue
331  * @class class for OtherHashAlgAndValue ASN.1 object
332  * @param {Array} params associative array of parameters
333  * @extends KJUR.asn1.ASN1Object
334  * @since jsrsasign 4.7.0 asn1cades 1.0.0
335  *
336  * @description
337  * This class provides ASN.1 encoder for
338  * <a href="https://tools.ietf.org/html/rfc5126#section-5.8.1">
339  * OtherHashAlgAndValue defined in RFC 5126 CAdES section 5.8.1</a>.
340  * <pre>
341  * OtherHashAlgAndValue ::= SEQUENCE {
342  *    hashAlgorithm   AlgorithmIdentifier,
343  *    hashValue       OtherHashValue }
344  * OtherHashValue ::= OCTET STRING
345  * </pre>
346  * Following properties can be apply to constructor arguments:
347  * <ul>
348  * <li>{String} alg - hash algorithm name for "hashAlgorithm" field</li>
349  * <li>{String} hash - hexadecimal string for "hashValue" field</li>
350  * </ul>
351  * 
352  * @example
353  * // specify by hash
354  * new KJUR.asn1.cades.OtherHashAlgAndValue({
355  *   alg: "sha256",
356  *   hash: "12abcd..."
357  * })
358  *
359  * // or specify by cert PEM or hex
360  * new KJUR.asn1.cades.OtherHashAlgAndValue({
361  *   alg: "sha256",
362  *   cert: "-----BEGIN..."
363  * })
364  * new KJUR.asn1.cades.OtherHashAlgAndValue({
365  *   alg: "sha256",
366  *   cert: "3082..."
367  * })
368  */
369 KJUR.asn1.cades.OtherHashAlgAndValue = function(params) {
370     var _Error = Error,
371 	_KJUR = KJUR,
372 	_KJUR_asn1 = _KJUR.asn1,
373 	_DERSequence = _KJUR_asn1.DERSequence,
374 	_DEROctetString = _KJUR_asn1.DEROctetString,
375 	_KJUR_asn1_x509 = _KJUR_asn1.x509,
376 	_AlgorithmIdentifier = _KJUR_asn1_x509.AlgorithmIdentifier,
377 	_KJUR_asn1_cades = _KJUR_asn1.cades,
378 	_OtherHashAlgAndValue = _KJUR_asn1_cades.OtherHashAlgAndValue;
379 
380     _OtherHashAlgAndValue.superclass.constructor.call(this);
381 
382     this.params = null;
383 
384     this.getEncodedHex = function() {
385 	var params = this.params;
386 
387 	if (params.alg == undefined)
388 	    throw new _Error("property 'alg' not specified");
389 
390 	if (params.hash == undefined && params.cert == undefined)
391 	    throw new _Error("property 'hash' nor 'cert' not specified");
392 
393 	var hHash = null;
394 	if (params.hash != undefined) {
395 	    hHash = params.hash;
396 	} else if (params.cert != undefined) {
397 	    if (typeof params.cert != "string")
398 		throw new _Error("cert not string");
399 
400 	    var hCert = params.cert;
401 	    if (params.cert.indexOf("-----BEGIN") != -1) {
402 		hCert = pemtohex(params.cert);
403 	    }
404 	    hHash = KJUR.crypto.Util.hashHex(hCert, params.alg);
405 	}
406 
407 	var a = [];
408 	a.push(new _AlgorithmIdentifier({name: params.alg}));
409 	a.push(new _DEROctetString({hex: hHash}));
410 	var seq = new _DERSequence({array: a});
411 	return seq.getEncodedHex();
412     };
413 
414     if (params != undefined) this.setByParam(params);
415 };
416 YAHOO.lang.extend(KJUR.asn1.cades.OtherHashAlgAndValue, KJUR.asn1.ASN1Object);

430  * OtherHashValue ::= OCTET STRING
431  * </pre>
432  *
433  * @example
434  * new KJUR.asn1.cades.OtherHashValue({hash: "12ab..."})

436  * new KJUR.asn1.cades.OtherHashValue({cert: "3081..."})
437  */
438 KJUR.asn1.cades.OtherHashValue = function(params) {
439     KJUR.asn1.cades.OtherHashValue.superclass.constructor.call(this);
440 
441     var _Error = Error,
442 	_KJUR = KJUR,
443 	_isHex = _KJUR.lang.String.isHex,
444 	_KJUR_asn1 = _KJUR.asn1,
445 	_DEROctetString = _KJUR_asn1.DEROctetString,
446 	_hashHex = _KJUR.crypto.Util.hashHex;
447     
448     this.params = null;
449 
450     this.getEncodedHex = function() {
451 	var params = this.params;
452 
453 	if (params.hash == undefined && params.cert == undefined) {
454 	    throw new _Error("hash or cert not specified");
455 	}
456 
457 	var hHash = null;
458 	if (params.hash != undefined) {
459 	    hHash = params.hash;
460 	} else if (params.cert != undefined) {
461 	    if (typeof params.cert != "string") {
462 		throw new _Error("cert not string");
463 	    }
464 	    var hCert = params.cert;
465 	    if (params.cert.indexOf("-----BEGIN") != -1) {
466 		hCert = pemtohex(params.cert);
467 	    }
468 	    hHash = KJUR.crypto.Util.hashHex(hCert, "sha1");
469 	}
470 	return (new _DEROctetString({hex: hHash})).getEncodedHex();
471     };
472 
473     if (params != undefined) this.setByParam(params);
474 };

477 /**
478  * class for RFC 5126 CAdES SignatureTimeStamp attribute<br/>
479  * @name KJUR.asn1.cades.SignatureTimeStamp
480  * @class class for RFC 5126 CAdES SignatureTimeStamp attribute
481  * @param {Array} params associative array of parameters
482  * @extends KJUR.asn1.cms.Attribute
483  * @since jsrsasign 4.7.0 asn1cades 1.0.0
484  * @description
485  * <pre>
486  * id-aa-signatureTimeStampToken OBJECT IDENTIFIER ::=
487  *    1.2.840.113549.1.9.16.2.14
488  * SignatureTimeStampToken ::= TimeStampToken
489  * </pre>
490  *
491  * @example
492  * // by TimeStampToken hex
493  * new KJUR.asn1.cades.SignatureTimeStamp({
494  *   attr: "timeStampToken",
495  *   tst: "3082..."})
496  *
497  * // by TimeStampToken or ASN1Object
498  * new KJUR.asn1.cades.SignatureTimeStamp({
499  *   attr: "timeStampToken",
500  *   tst: new TimeStampToken(...)})
501  *
502  * // by TimeStampResponse hex
503  * new KJUR.asn1.cades.SignatureTimeStamp({
504  *   attr: "timeStampToken",
505  *   res: "3082..."})
506  *
507  * // by TimeStampToken or ASN1Object
508  * new KJUR.asn1.cades.SignatureTimeStamp({
509  *   attr: "timeStampToken",
510  *   res: new TimeStampResponse(...)})
511  */
512 KJUR.asn1.cades.SignatureTimeStamp = function(params) {
513     var _Error = Error,
514 	_KJUR = KJUR,
515 	_isHex = _KJUR.lang.String.isHex,
516 	_KJUR_asn1 = _KJUR.asn1,
517 	_ASN1Object = _KJUR_asn1.ASN1Object,
518 	_KJUR_asn1_x509 = _KJUR_asn1.x509,
519 	_KJUR_asn1_cades = _KJUR_asn1.cades;
520 
521     _KJUR_asn1_cades.SignatureTimeStamp.superclass.constructor.call(this);
522     this.typeOid = "1.2.840.113549.1.9.16.2.14";
523     this.params = null;
524 
525     this.getValueArray = function() {
526 	var params = this.params;
527 
528 	if (params.tst != undefined) {
529 	    if (_isHex(params.tst)) {
530 		var dTST = new _ASN1Object();
531 		dTST.hTLV = params.tst;
